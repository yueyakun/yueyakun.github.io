RAS+AES混合加密Vue端实现

kun 2021-01-17 API加密

分下面三个日志记录：

这里是第二篇：RAS+AES混合加密Vue端实现

# 请求和返回结果的拦截处理

前端请求我采用的是 Axios 框架，Axios的 request 拦截器（请求发出前触发） 和

response 拦截器（收到结果后触发）组件能很方便地对请求和返回结果做全局处理。

# 遇到个问题

我本来打算在 request 拦截器中进行加密和签名操作，在response 拦截器中进行解密返回结果的操作。

看起来很完美是吧？但是，，，有一个非常棘手的问题：在 request 拦截器中生成的 aesKey 要怎样传到 response 拦截器中呢？我想到了下面几个方案，但是都被一一否定了：

  • 把 aesKey 生成全局变量？不行，并发的时候会有问题。

  • 把 aesKey 放到 ThreaLocal 找那个向后传递？不行，JS没有 ThreaLocal 这种东西，自己实现一个代价太低。

  • 把 aesKey 放到 sessionStorage 中？不行，不安全。

  • 把 aesKey 从后端再传回来？不行，传回来还得加密，得在维护一对 RAS 秘钥，太蠢了。

就在我以为没办法了，只能从后端加密传回来的时候，我在 Axios 的文档中找到了解决方案：

请求的 config 中除了能定义 url、method、header 等参数外还能定义一个叫 stransformResponse 的回调函数，这个函数会在返回结果进入 response 拦截器之前执行！简直完美！哈哈，看来有问题还是得看官方文档啊！

# RAS 和 AES 组件

RAS 我采用的是 jsencrypt 组件，npm 安装命令如下：

npm install jsencrypt --save

AES 我采用的是 jsencrypt 组件，npm 安装命令如下：

npm install crypto-js --save

# 详细处理步骤

request 拦截器要做的事情主要有以下几步：

  1. 生成 AES 秘钥（aesKey）；
  2. 加密请求体重的数据得到加密结果（encryptedBody），替换原来的明文 body；
  3. 生成时间戳 （timestamp）和随机数（nonce）
  4. 将请求参数 requestParam （路径传的参数）、encryptedBody、aesKey、timestamp（时间戳）、随机数（nonce） 做签名处理，得到签名值 sign；
  5. 请求头中放入 sign、timestamp、nonce 和 encryptedAesKey；
  6. 定义 stransformResponse 回调函数，在里面实现解密逻辑；
  7. 发起请求。

# 代码实现

request 拦截器代码如下：

axios.interceptors.request.use(
  config => {
    //生成aesKey
    let aesKey = aesUtil.getAesKey()
    //加密data
    let encryptData = aesUtil.encrypt(config.data,aesKey)
    //替换明文body
    config.data = encryptData
    //签名
    let timestamp = new Date().getTime()
    let nonce = uuidv4()
    let sign = aesUtil.sign(aesKey,encryptData,timestamp,nonce,config.params)
    //加密aesKey
    let encryptedAesKey = rasUtil.encrypt(aesKey)
    //设置请求头
    config.headers['X_EAK'] = encryptedAesKey
    config.headers['X_SIGN'] = sign
    config.headers['X_TIMESTAMP'] = timestamp
    config.headers['X_NONCE'] = nonce
    config.transformResponse = [function (dataStr) {
      //此回调先于response拦截器执行，在这里解密简直完美
      let data = JSON.parse(dataStr)
      if (data.encrypted) {
        let decrypt = aesUtil.decrypt(data.data, aesKey)
        data.data = decrypt
      }
      return data;
    }]
    return config
  },error => {
    // 对请求错误做些什么
    return Promise.reject(error);
  }
)

aesUtil 代码如下：

import CryptoJS from "crypto-js";

const chars = 'ABCDEFGHJKMNPQRSTWXYZabcdefhijkmnprstwxyz2345678';
/****默认去掉了容易混淆的字符oOLl,9gq,Vv,Uu,I1****/

export default {

  getAesKey(len) {
    len = len || 32;
    let maxPos = chars.length;
    let keyStr = '';
    for (let i = 0; i < len; i++) {
      keyStr += chars.charAt(Math.floor(Math.random() * maxPos));
    }
    // console.log("生成aesKye：",keyStr)
    return keyStr;
  },

  //aes加密
  encrypt(data, key) {
    // console.log("aes加密前：",data)
    if (!data){
      return data
    }
    let encrypted = "";
    let dataStr = '';
    if (typeof data == "string") {
      // console.log("typeof：string")
      dataStr = CryptoJS.enc.Utf8.parse(data);
      encrypted = CryptoJS.AES.encrypt(dataStr, CryptoJS.enc.Utf8.parse(key), {//这是一个大坑，utf8字符串的密钥必须转换城一个WordArray对象才会被当成密码，否则会被当成生成密码的口令
        mode: CryptoJS.mode.ECB,
        padding: CryptoJS.pad.Pkcs7
      });
    } else if (typeof data == "object") {
      // console.log("typeof：object")
      //对象格式的转成json字符串
      dataStr = JSON.stringify(data);
      // console.log("aes加密前：",dataStr)
      encrypted = CryptoJS.AES.encrypt(dataStr, CryptoJS.enc.Utf8.parse(key), {
        mode: CryptoJS.mode.ECB,
        padding: CryptoJS.pad.Pkcs7
      });
    }
    //这里有个坑，加密结果是个对象，里面有很多属性。。。要取 ciphertext
    let encryptedHexStr = CryptoJS.enc.Hex.parse(encrypted.ciphertext.toString());
    // console.log("传给后端：",CryptoJS.enc.Base64.stringify(encryptedHexStr))
    return CryptoJS.enc.Base64.stringify(encryptedHexStr);//返回base64格式
  },

  // aes解密
  decrypt(encrypted, key) {
    // console.log("ase解密前：",encrypted)
    let decrypt = CryptoJS.AES.decrypt(encrypted, CryptoJS.enc.Utf8.parse(key), {
      mode: CryptoJS.mode.ECB,
      padding: CryptoJS.pad.Pkcs7
    });
    // console.log(decrypt.toString(CryptoJS.enc.Utf8))
    let decryptedObj = JSON.parse(decrypt.toString(CryptoJS.enc.Utf8));
    // console.log("ase解密后：",decryptedObj)
    return decryptedObj;
  },

  sign(aesKey, encryptData, timestamp, nonce, params) {
    // console.log("签名参数：","aesKey:",aesKey,"encryptData:",encryptData,"timestamp:",timestamp,"nonce:",nonce,"params:",params)
    //排序所有属性
    let allParams = {aesKey: aesKey, timestamp: timestamp, nonce: nonce, body: encryptData}
    if (params) {
      for (let p in params) {
        if (params.hasOwnProperty(p) && (!allParams.hasOwnProperty(p)))
          allParams[p] = params[p];
      }
    }
    //将值拼接到一起
    let values = []
    Object.keys(allParams).sort().map(key=>{
      values.push(allParams[key])
    })
    let valuesStr = values.join('')
    let sign = CryptoJS.MD5(valuesStr).toString()
    // console.log("签名结果：",sign)
    return sign
  }
}

rasUtil 代码如下：

import {JSEncrypt} from "jsencrypt";

const publicKey = 'XXXXX';

export default {

  encrypt(word) {
    // console.log("rsa加密前：",word)
    let encrypt = new JSEncrypt();

# 还有几个小坑

第一个：

生成的 aesKey 是 32 位的 utf8 字符串，直接传入 crypto-js 的 encrypt 方法的话会被当做口令而不是秘钥，crypto-js 会根据传入的口令生成一组秘钥、偏移量（iv）、盐（salt）。而且即使口令一样，每次生成的秘钥等也是不一样的。所以加密时需要把生成 aesKey 转换城一个 WordArray 对象才会被当成密码，否则会被当成生成密码的口令。

WordArray 和其他类型之间的转换关系如下：

//创建WordArray
var wordArray = CryptoJS.lib.WordArray.create();//创建一个空的 WordArray对象
//WordArray转字符串
var string = wordArray.toString();//默认CryptoJS.enc.Hex，即16进制字符串
var string = wordArray.toString(CryptoJS.enc.Utf8);//utf-8字符串
//十六进制字符串转WordArray
var wordArray = CryptoJS.enc.Hex.parse(hexString);
//utf8字符串转WordArray
var wordArray = CryptoJS.enc.Utf8.parse(utf8String);
//WordArray转Base64字符串
var base64String = CryptoJS.enc.Base64.stringify(wordArray);
//Base64字符串转WordArray
var wordArray = CryptoJS.enc.Base64.parse(base64String);

第二个：

加密的结果 encrypted 是一个对象，实际的加密结果在 ciphertext 属性中，要用 encrypted.ciphertext.toString() 方式取加密后的密文，如果直接 encrypted.toString() 的话得到的是一个以 U2FsdGVkX1 开头的 OpenSSL 格式的结果。encrypted 属性列表如下：

  • key：秘钥（传入的秘钥如果是口令的话会自动派生一个key）
  • iv：传入的偏移量（传入的秘钥如果是口令的话会自动派生iv字段）
  • salt：盐（传入的秘钥如果是口令的话会自动派生salet字段）
  • ciphertext：加密后的密文

第三个：

调用 decrypt 方法解密时得到的是一个 WordArray 对象，转换成正常对象的方式是：JSON.parse(decrypt.toString(CryptoJS.enc.Utf8))。
