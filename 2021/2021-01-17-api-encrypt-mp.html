# 代码实现

有了 Vue 端实现的基础，在小程序端的实现就简单多了，签名、加解密逻辑和 Vue 端一样。不同的是小程序请求都需要通过 wx.request 发送，所以需要对 wx.request 做一个封装，直接上代码。

aesUtil.js 代码如下：

import CryptoJS from "crypto-js";

/** 用来生成aesKey的码表，去掉了容易混淆的字符oOLl,9gq,Vv,Uu,I1 */
const chars = 'ABCDEFGHJKMNPQRSTWXYZabcdefhijkmnprstwxyz2345678';

export default {

  getAesKey(len) {
    len = len || 32;

    let maxPos = chars.length;
    let keyStr = '';
    for (let i = 0; i < len; i++) {
      keyStr += chars.charAt(Math.floor(Math.random() * maxPos));
    }
    return keyStr;
  },

  //aes加密
  encrypt(data, key) {
    console.log("aes加密前：",data)
    if (!data){
      return data
    }
    let encrypted = "";
    let dataStr = '';
    if (typeof data == "string") {
      console.log("typeof：string")
      dataStr = CryptoJS.enc.Utf8.parse(data);
      encrypted = CryptoJS.AES.encrypt(dataStr, CryptoJS.enc.Utf8.parse(key), {//这是一个大坑，utf8字符串的密钥必须转换城一个WordArray对象才会被当成密码，否则会被当成生成密码的口令
        mode: CryptoJS.mode.ECB,
        padding: CryptoJS.pad.Pkcs7
      });
    } else if (typeof data == "object") {
      console.log("typeof：object")
      //对象格式的转成json字符串
      dataStr = JSON.stringify(data);
      encrypted = CryptoJS.AES.encrypt(dataStr, CryptoJS.enc.Utf8.parse(key), {
        mode: CryptoJS.mode.ECB,
        padding: CryptoJS.pad.Pkcs7
      });
    }
    //这里有个坑，加密结果是个对象，里面有很多属性。。。
    let encryptedHexStr = CryptoJS.enc.Hex.parse(encrypted.ciphertext.toString());
    return CryptoJS.enc.Base64.stringify(encryptedHexStr);//
  },

  // aes解密
  decrypt(encrypted, key) {
    let decrypt = CryptoJS.AES.decrypt(encrypted, CryptoJS.enc.Utf8.parse(key), {
      mode: CryptoJS.mode.ECB,
      padding: CryptoJS.pad.Pkcs7
    });
    let decryptedObj = JSON.parse(decrypt.toString(CryptoJS.enc.Utf8));
    console.log("ase解密后：",decryptedObj)
    return decryptedObj;
  },

  sign(aesKey, encryptData, timestamp, nonce, params) {
    //排序所有属性
    let allParams = {aesKey: aesKey, timestamp: timestamp, nonce: nonce, body: encryptData}
    if (params) {
      for (let p in params) {
        if (params.hasOwnProperty(p) && (!allParams.hasOwnProperty(p)))
          allParams[p] = params[p];
      }
    }
    //将值拼接到一起
    let values = []
    Object.keys(allParams).sort().map(key=>{
      values.push(allParams[key])
    })
    let valuesStr = values.join('')
      console.log("签名字符串：",valuesStr)
    let sign = CryptoJS.MD5(valuesStr).toString()
    console.log("签名结果：",sign)
    return sign
  }
}

rasUtil.js 代码如下：

import {JSEncrypt} from "jsencrypt.js";

const publicKey = 'xxx';

const hexDigits = "0123456789abcdef";

export default {

    encrypt(word) {
        console.log("rsa加密前：", word)
        let encrypt = new JSEncrypt();
        encrypt.setPublicKey(publicKey);
        let encrypted = encrypt.encrypt(word)
        console.log("rsa加密后：", encrypted)
        return encrypted;
    },
    uuid() {

        let s = [];

        for (let i = 0; i < 36; i++) {
            s[i] = hexDigits.substr(Math.floor(Math.random() * 0x10), 1);
        }

        s[14] = "4";  // bits 12-15 of the time_hi_and_version field to 0010
        s[19] = hexDigits.substr((s[19] & 0x3) | 0x8, 1);  // bits 6-7 of the clock_seq_hi_and_reserved to 01
        s[8] = s[13] = s[18] = s[23] = "-";

        let uuid = s.join("");

        return uuid;
    }
}

api.js 代码如下：

import aesUtil from 'aesUtil'
import rasUtil from 'rasUtil'
// 自动根据版本切换接口请求地址
const {miniProgram: {envVersion}} = wx.getAccountInfoSync();
let baseUrl = '';
switch (envVersion) {
    case 'develop':
        baseUrl = 'http://localhost:8083';//其实不能用localhos，开发环境允许时IP地址但是不能是本机ip
        break;
    case 'trial':
        baseUrl = 'https://xxx.xxx.com/test';
        break;
    case 'release':
        baseUrl = 'https:///xxx.xxx.com/api';
        break;
    default:
        baseUrl = 'https:///xxx.xxx.com/api';
        break;
}

export function request (options) {
    return new Promise((resolve, reject) => {
        // 拼接url
        let url = baseUrl + options.url
        if (options.params) {
            url = url + '?'
            for (let key in options.params) {
                url = `${url}${key}=${options.params[key]}&`
            }
            url = url.substring(0,url.length-1)
        }
        options.url = url
        // 生成aesKey
        let aesKey = aesUtil.getAesKey()
        //加密data
        let encryptData = aesUtil.encrypt(options.data, aesKey)
        //签名
        let timestamp = new Date().getTime()
        let nonce = rasUtil.uuid()
        let sign = aesUtil.sign(aesKey, encryptData, timestamp, nonce, options.params)
        //加密aesKey
        let encryptedAesKey = rasUtil.encrypt(aesKey)
        //设置请求头
        options.header = options.header || {}
        Object.assign(options.header, {
            X_EAK: encryptedAesKey,
            X_SIGN: sign,
            X_TIMESTAMP: timestamp,
            X_NONCE: nonce
        })
        console.log("options:",options)
        wx.showLoading({
            title: '加载中...',
        })
        wx.request({
            ...options,
            success: function (res) {
                wx.hideLoading()
                if (res.statusCode === 200&&res.data.code===200) {
                    if (res.data && res.data.encrypted) {
                        let decrypt = aesUtil.decrypt(res.data.data, aesKey)
                        res.data.data = decrypt
                    }
                    resolve(res.data)
                } else {
                    reject(res.data)
                }
            },
            fail: function (res) {
                wx.hideLoading()
                reject(res.data)
            }
        })
    })
}

# 一个 Nginx 问题

这个问题是我把服务部署到线上的时候遇到的。线上使用了 Nginx，然后发现后端接不到请求头中的 X_SIGN、X_TIMESTAMP、X_NONCE 等自定义的请求头。最后发现原因是这样的：

nginx对header name的字符做了限制，默认 underscores_in_headers 为off，表示如果header name中包含下划线，则忽略掉。而我的自定义header中恰巧有下划线变量。

所以，后续再碰到类似情况，要么header中自定义变量名不要用下划线，要么在nginx.conf中加上underscores_in_headers on配置。

# 不足

严格的 API 验签加密架构肯定会给每个客户端分配 appToken 和 appSecret，appToken 用来网络传输，标志调用者身份。 appSecret 不在网络中传输，只用来签名加密。后端根据传过来的 appToken 查询到对应的 appSecret 然后验签解密。 但是我不想搞这么复杂，所以只给前端了一个 RSA 公钥，相当于 appSecret 吧。这样简化了验签时的步骤，但是 RSA 公钥一旦泄露，别人只要熟悉你的验签逻辑就可以随便调用你的接口了。
